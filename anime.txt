    <script>
        // Create animated particles
        const particlesContainer = document.getElementById('particles');
        for (let i = 0; i < 50; i++) {
            const particle = document.createElement('div');
            particle.className = 'particle';
            particle.style.left = Math.random() * 100 + '%';
            particle.style.animationDelay = Math.random() * 15 + 's';
            particle.style.animationDuration = (Math.random() * 10 + 10) + 's';
            particlesContainer.appendChild(particle);
        }

        // Smooth scrolling
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                const target = document.querySelector(this.getAttribute('href'));
                if (target) {
                    target.scrollIntoView({ behavior: 'smooth' });
                }
            });
        });

        // Add interactive hover effects to service cards
        document.querySelectorAll('.service-card').forEach(card => {
            card.addEventListener('mousemove', (e) => {
                const rect = card.getBoundingClientRect();
                const x = e.clientX - rect.left;
                const y = e.clientY - rect.top;
                
                card.style.background = `
                    radial-gradient(circle at ${x}px ${y}px, 
                    rgba(0, 255, 255, 0.15), 
                    rgba(255, 255, 255, 0.05))
                `;
            });
            
            card.addEventListener('mouseleave', () => {
                card.style.background = 'rgba(255, 255, 255, 0.05)';
            });
        });

        // Chatbot functionality
        const chatbot = document.getElementById('chatbot');
        const chatMessages = document.getElementById('chatMessages');
        const chatInput = document.getElementById('chatInput');
        const typingIndicator = document.getElementById('typingIndicator');

        function toggleChat() {
            chatbot.classList.toggle('active');
        }

        function addMessage(message, isBot = false) {
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${isBot ? 'bot' : 'user'}`;
            messageDiv.textContent = message;
            chatMessages.appendChild(messageDiv);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function showTyping() {
            typingIndicator.classList.add('active');
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        function hideTyping() {
            typingIndicator.classList.remove('active');
        }

        function getBotResponse(userMessage) {
            const msg = userMessage.toLowerCase();
            
            if (msg.includes('repair cost') || msg.includes('price') || msg.includes('cost')) {
                return "üí∞ Our AI diagnostics start at $29! Screen repairs range from $79-$299 depending on your device. Upload a photo and I'll give you an instant quote!";
            } else if (msg.includes('repair time') || msg.includes('how long')) {
                return "‚ö° Our quantum repair technology completes most repairs in just 15-30 minutes! Complex repairs take 2-4 hours max. We're FAST!";
            } else if (msg.includes('warranty') || msg.includes('guarantee')) {
                return "üõ°Ô∏è Every repair comes with our AI-backed lifetime warranty! If anything goes wrong, we'll fix it free. Zero human error = 100% confidence!";
            } else if (msg.includes('track') || msg.includes('status') || msg.includes('order')) {
                return "üìç You can track your repair in real-time! Just enter your ticket number in the Track Repair section. Our AI keeps you updated every step of the way!";
            } else if (msg.includes('hello') || msg.includes('hi') || msg.includes('hey')) {
                return "üëã Hello! I'm Gemini, your AI repair expert. Ready to fix your device with cutting-edge technology. What can I help you with?";
            } else if (msg.includes('thank')) {
                return "‚ú® You're welcome! Jay's Shop is here 24/7 with AI-powered support. Need anything else?";
            } else if (msg.includes('phone') || msg.includes('iphone') || msg.includes('android')) {
                return "üì± We repair all phones! iPhone, Samsung, Google Pixel, OnePlus - you name it! Our nano-tech robots handle every brand with precision. What issue are you experiencing?";
            } else {
                return "ü§ñ Great question! I can help you with repair costs, timing, warranty info, and device issues. Or you can speak with our human team at (555) JAYS-SHOP. How can I assist you today?";
            }
        }

        function sendMessage() {
            const message = chatInput.value.trim();
            if (!message) return;

            addMessage(message, false);
            chatInput.value = '';

            showTyping();
            
            setTimeout(() => {
                hideTyping();
                const response = getBotResponse(message);
                addMessage(response, true);
            }, 1000 + Math.random() * 1000);
        }

        function quickMessage(type) {
            const messages = {
                'repair cost': 'What are your repair costs?',
                'repair time': 'How long do repairs take?',
                'warranty': 'Tell me about your warranty',
                'track order': 'How do I track my repair?'
            };
            
            chatInput.value = messages[type];
            sendMessage();
        }

        function handleKeyPress(event) {
            if (event.key === 'Enter') {
                sendMessage();
            }
        }
